HTTP Request Basic Authentication credentials must use property placeholders

When Basic Authentication is used, the 'username' and 'password' attributes in must reference property placeholders (e.g., ${api.username}, ${api.password}) instead of hardcoded values. Hardcoded credentials are a critical security vulnerability.

Noncompliant Code

<mule xmlns="http://www.mulesoft.org/schema/mule/core"
      xmlns:http="http://www.mulesoft.org/schema/mule/http">
  <http:request-config name="HTTP_Request_config">
    <http:request-connection host="api.example.com">
      <http:authentication>
        <http:basic-authentication username="admin" password="secret123"/>
      </http:authentication>
    </http:request-connection>
  </http:request-config>
</mule>

Compliant Code

<mule xmlns="http://www.mulesoft.org/schema/mule/core"
      xmlns:http="http://www.mulesoft.org/schema/mule/http">
  <http:request-config name="HTTP_Request_config">
    <http:request-connection host="api.example.com">
      <http:authentication>
        <http:basic-authentication username="${api.username}" password="${api.password}"/>
      </http:authentication>
    </http:request-connection>
  </http:request-config>
</mule>
